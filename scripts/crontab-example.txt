# Optimized Scraper Cron Jobs
# Add these to your crontab: crontab -e

# Quick scraping every 6 hours (4 times per day)
0 */6 * * * cd /path/to/plan2fund-nextgen && node scripts/optimized-scraper-cron.js quick >> logs/scraper-quick.log 2>&1

# Full scraping daily at 2 AM
0 2 * * * cd /path/to/plan2fund-nextgen && node scripts/optimized-scraper-cron.js full >> logs/scraper-full.log 2>&1

# Cleanup old data daily at 3 AM (after full scrape)
0 3 * * * cd /path/to/plan2fund-nextgen && node scripts/optimized-scraper-cron.js cleanup >> logs/scraper-cleanup.log 2>&1

# Weekly stats report (optional)
0 9 * * 1 cd /path/to/plan2fund-nextgen && node scripts/optimized-scraper-cron.js stats >> logs/scraper-stats.log 2>&1
